// Format timestamp to human-readable format (12-hour with AM/PM) function formatTimeLabel(dateStr) { try { const date = new Date(dateStr + 'Z'); // Add Z to parse as UTC return date.toLocaleString('en-US', { month: 'short', // Apr day: 'numeric', // 25 hour: '2-digit', // 04 minute: '2-digit', // 40 hour12: true, // Use 12-hour format timeZone: 'Asia/Kolkata' // IST timezone (change if needed) }); } catch (err) { return dateStr; // Fallback } }
